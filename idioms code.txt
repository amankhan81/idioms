            <script>
                // Variable to hold the currently playing audio element
                let currentAudio = null;

                /**
                 * Converts an idiom string into a snake_case filename, adding the .mp3 extension.
                 * @param {string} idiom - The idiom string.
                 * @returns {string} The suggested filename.
                 */
                function createFilename(idiom) {
                    return idiom.toLowerCase().replace(/\s+/g, '_') + '.mp3';
                }

                /**
                 * Plays the actual audio file associated with the idiom, stopping any previous audio.
                 * @param {string} filename - The name of the audio file (e.g., 'over_the_moon.mp3').
                 * @param {string} idiom - The idiom being played.
                 */
                function playAudio(filename, idiom) {
                    console.log(`Attempting to play audio for: "${idiom}". File: ${filename}`);
                    
                    // 1. Stop and reset the previous audio track, if one exists
                    if (currentAudio && !currentAudio.paused) {
                        currentAudio.pause();
                        currentAudio.currentTime = 0; // Reset playback position
                        console.log("Previous audio stopped.");
                    }
                    
                    // 2. Create the new audio object
                    const newAudio = new Audio(filename);
                    
                    // 3. Store the new audio object globally
                    currentAudio = newAudio;

                    // 4. Start playback
                    newAudio.play().catch(e => {
                        // Log errors if audio fails to load or play
                        console.error(`Error playing audio for ${idiom} (file: ${filename}). Ensure the file exists and is correctly named.`, e);
                        // Optionally reset currentAudio if playback fails immediately
                        currentAudio = null; 
                    });
                }
            </script>
